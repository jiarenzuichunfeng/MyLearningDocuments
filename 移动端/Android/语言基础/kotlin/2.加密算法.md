## 凯撒加密

* 字符做偏移

  * 

  * ```kotlin
    class CaesarCrypt {
        /**
         *  凯撒加密
         *  @param input 明文
         *  @param key 私钥
         */
        fun encrypt(input: String,key: Int) : String {
            val charArray = input.toCharArray()
            return with(StringBuffer()){
                charArray.forEach {
                    val c = it
                    var ascii = c.code
                    ascii += key
                    val result = ascii.toChar()
                    append(result)
                }
                toString()
            }
        }
        /**
         * 解密
         * @param input 加密过后的字符串
         * @param key 私钥
         */
        fun decrypt(input: String,key: Int) : String {
            val charArray = input.toCharArray()
            return with(StringBuffer()){
                charArray.forEach {
                    val c = it
                    var ascii = c.code
                    ascii -= key
                    val result = ascii.toChar()
                    append(result)
                }
                toString()
            }
        }
    
    }
    ```

## 对称加密

* DES 

  1. 创建cipher对象

  2. 初始化cipher

  3. 加密/解密

  4. ```kotlin
     // DES 加密工具类
     object DESCrypt {
     
         /**
          * DES 加密方法
          * @param input 要加密的明文字符串
          * @param password 加密密钥（DES 要求密钥长度必须是 8 字节）
          * @return 加密后的 Base64 编码字符串
          */
         fun encrypt(input: String, password: String): String {
             // 获取 DES 加密算法的 Cipher 实例（用于执行加密操作）
             val cipher = Cipher.getInstance("DES")
     
             // 获取 DES 算法的密钥工厂实例（用于生成密钥）
             val kf = SecretKeyFactory.getInstance("DES")
     
             // 根据密码字节数组创建 DES 密钥规格（定义密钥规则）
             val keySpec = DESKeySpec(password.toByteArray())
     
             // 通过密钥工厂和密钥规格生成真正的 DES 密钥对象
             val key: Key = kf.generateSecret(keySpec)
     
             // 初始化 Cipher 为【加密模式】，并传入密钥
             cipher.init(Cipher.ENCRYPT_MODE, key)
     
             // 执行加密操作：将明文字节数组加密成密文字节数组
             val encrypt = cipher.doFinal(input.toByteArray())
     
             // 将加密后的字节数组进行 Base64 编码，转为字符串返回（方便传输存储）
             return Base64.getEncoder().encodeToString(encrypt)
         }
     
         /**
          * DES 解密方法
          * @param input 加密后的 Base64 字符串
          * @param password 解密密钥（必须和加密密钥一致）
          * @return 解密后的明文字符串
          */
         fun decrypt(input: String, password: String): String {
             // 获取 DES 加密算法的 Cipher 实例（用于执行解密操作）
             val c = Cipher.getInstance("DES")
     
             // 获取 DES 算法的密钥工厂实例
             val kf = SecretKeyFactory.getInstance("DES")
     
             // 根据密码字节数组创建 DES 密钥规格
             val keySpec = DESKeySpec(password.toByteArray())
     
             // 生成 DES 密钥对象
             val key: Key = kf.generateSecret(keySpec)
     
             // 初始化 Cipher 为【解密模式】，传入密钥
             c.init(Cipher.DECRYPT_MODE, key)
     
             // 先对输入的 Base64 字符串解码，再执行解密得到明文字节数组
             val encrypt = c.doFinal(Base64.getDecoder().decode(input))
     
             // 将明文字节数组转为字符串返回
             return String(encrypt)
         }
     }
     ```

* AES

  * ```kotlin
    /**
     * Kotlin AES 对称加密工具类
     * 算法：AES/CBC/PKCS5Padding
     * 特点：安全、通用、可跨平台（Java/Android/后端）
     */
    object AesUtils {
        // 加密算法固定配置
        private const val ALGORITHM = "AES"
        private const val TRANSFORMATION = "AES/CBC/PKCS5Padding"
        private const val KEY_SIZE = 256 // 可选 128 / 192 / 256（256最安全）
        private const val IV_SIZE = 16   // AES IV 固定 16 字节
    
        /**
         * 生成随机 AES 密钥（256位）
         */
        fun generateSecretKey(): SecretKey {
            val keyGen = KeyGenerator.getInstance(ALGORITHM)
            keyGen.init(KEY_SIZE)
            return keyGen.generateKey()
        }
    
        /**
         * 生成随机 IV（向量）
         */
        fun generateIv(): IvParameterSpec {
            val iv = ByteArray(IV_SIZE)
            java.security.SecureRandom().nextBytes(iv)
            return IvParameterSpec(iv)
        }
    
        /**
         * 加密
         * @param data 明文
         * @param secretKey 密钥
         * @param iv 向量
         * @return Base64 加密字符串
         */
        fun encrypt(data: String, secretKey: SecretKey, iv: IvParameterSpec): String {
            val cipher = Cipher.getInstance(TRANSFORMATION)
            cipher.init(Cipher.ENCRYPT_MODE, secretKey, iv)
            val encryptedBytes = cipher.doFinal(data.toByteArray(Charsets.UTF_8))
            return Base64.getEncoder().encodeToString(encryptedBytes)
        }
    
        /**
         * 解密
         * @param encryptedData Base64 加密字符串
         * @param secretKey 密钥
         * @param iv 向量
         * @return 明文
         */
        fun decrypt(encryptedData: String, secretKey: SecretKey, iv: IvParameterSpec): String {
            val cipher = Cipher.getInstance(TRANSFORMATION)
            cipher.init(Cipher.DECRYPT_MODE, secretKey, iv)
            val decryptedBytes = cipher.doFinal(Base64.getDecoder().decode(encryptedData))
            return String(decryptedBytes, Charsets.UTF_8)
        }
    
        // ---------------- 密钥/IV 与 Base64 互转（方便存储传输） ----------------
        fun secretKeyToBase64(key: SecretKey): String =
            Base64.getEncoder().encodeToString(key.encoded)
    
        fun base64ToSecretKey(base64Key: String): SecretKey =
            SecretKeySpec(Base64.getDecoder().decode(base64Key), ALGORITHM)
    
        fun ivToBase64(iv: IvParameterSpec): String =
            Base64.getEncoder().encodeToString(iv.iv)
    
        fun base64ToIv(base64Iv: String): IvParameterSpec =
            IvParameterSpec(Base64.getDecoder().decode(base64Iv))
    }

    /**
     * RSA 非对称加密工具类（支持长文本分段加解密）
     * 核心规则：
     * 1. 公钥加密 → 私钥解密
     * 2. 私钥加密 → 公钥解密
     * 3. 加密/解密必须使用同一套密钥对
     * 4. 因 RSA 有长度限制，大文本必须分段处理
     */
    object RSACrypt {
    
        /** 加密算法 */
        val ALGORITHM = "RSA"
    
        /** RSA 单次最大加密长度（根据密钥长度调整） */
        val ENCRYPT_MAX_SIZE = 373
    
        /** RSA 单次最大解密长度（与加密长度对应） */
        val DECRYPT_MAX_SIZE = 384
    
        /**
         * 【私钥加密】
         * 使用私钥对明文字符串进行分段加密
         * @param input 明文字符串
         * @param privateKey 私钥
         * @return Base64 编码后的加密字符串
         */
        fun encryptPrivate(input: String, privateKey: PrivateKey): String? {
            // 将明文字符串转换为字节数组
            val batrArray = input.toByteArray()
            // 临时存储每一段加密后的结果
            var temp: ByteArray? = null
            // 偏移量：记录当前处理到字节数组的哪个位置
            var offset = 0
            // 输出流：拼接所有分段加密后的数据
            val bos: ByteArrayOutputStream = ByteArrayOutputStream()
    
            // 获取 RSA 加密实例
            val cipher = Cipher.getInstance(ALGORITHM)
            // 初始化为 私钥加密模式
            cipher.init(Cipher.ENCRYPT_MODE, privateKey)
    
            // 循环分段加密，直到所有数据处理完成
            while (batrArray.size - offset > 0) {
                // 计算本次需要加密的长度：最大长度 或 剩余长度
                val len = if (batrArray.size - offset >= ENCRYPT_MAX_SIZE) ENCRYPT_MAX_SIZE else batrArray.size - offset
                // 对当前分段执行加密
                temp = cipher.doFinal(batrArray, offset, len)
                // 将加密结果写入输出流
                bos.write(temp)
                // 偏移量向后移动，准备下一段
                offset += len
            }
    
            // 关闭输出流
            bos.close()
            // 将最终加密的字节数组进行 Base64 编码，返回字符串
            return Base64.getEncoder().encodeToString(bos.toByteArray())
        }
    
        /**
         * 【公钥加密】
         * 使用公钥对明文字符串进行分段加密
         * @param input 明文字符串
         * @param publicKey 公钥
         * @return Base64 编码后的加密字符串
         */
        fun encryptPublic(input: String, publicKey: PublicKey): String? {

            // 临时存储每一段加密后的结果

            // 输出流：拼接所有分段加密后的数据
            var bos: ByteArrayOutputStream = ByteArrayOutputStream()
    
            // 获取 RSA 加密实例
            val cipher = Cipher.getInstance(ALGORITHM)
            // 初始化为 公钥加密模式
            cipher.init(Cipher.ENCRYPT_MODE, publicKey)
    
            // 循环分段加密，直到所有数据处理完成
            while (batrArray.size - offset > 0) {
                // 计算本次需要加密的长度
                val len = if (batrArray.size - offset >= ENCRYPT_MAX_SIZE) ENCRYPT_MAX_SIZE else batrArray.size - offset
                // 对当前分段执行加密
                temp = cipher.doFinal(batrArray, offset, len)
                // 将加密结果写入输出流
                bos.write(temp)
                // 偏移量后移
                offset += len
            }
    
            bos.close()
            // Base64 编码后返回
            return Base64.getEncoder().encodeToString(bos.toByteArray())
        }
    
        /**
         * 【私钥解密】
         * 用于解密【公钥加密】的内容
         * @param input Base64 加密字符串
         * @param privateKey 私钥
         * @return 解密后的明文字符串
         */
        fun decryptPrivate(input: String, privateKey: PrivateKey): String {
            // 将 Base64 密文字符串解码为原始加密字节数组
            val batrArray = Base64.getDecoder().decode(input.toByteArray())
            var temp: ByteArray? = null
            var offset = 0
            var bos: ByteArrayOutputStream = ByteArrayOutputStream()
    
            // 获取 RSA 解密实例
            val cipher = Cipher.getInstance(ALGORITHM)
            // 初始化为 私钥解密模式
            cipher.init(Cipher.DECRYPT_MODE, privateKey)
    
            // 循环分段解密
            while (batrArray.size - offset > 0) {
                // 计算本次解密长度
                val len = if (batrArray.size - offset >= DECRYPT_MAX_SIZE) DECRYPT_MAX_SIZE else batrArray.size - offset
                // 执行分段解密
                temp = cipher.doFinal(batrArray, offset, len)
                // 写入输出流拼接结果
                bos.write(temp)
                // 偏移量后移
                offset += len
            }
    
            bos.close()
            // 将解密后的字节数组转回明文字符串
            return String(bos.toByteArray())
        }
    
        /**
         * 【公钥解密】
         * 用于解密【私钥加密】的内容
         * @param input Base64 加密字符串
         * @param publicKey 公钥
         * @return 解密后的明文字符串

            // 循环分段解密
            while (batrArray.size - offset > 0) {
                // 计算本次解密长度
                val len = if (batrArray.size - offset >= DECRYPT_MAX_SIZE) DECRYPT_MAX_SIZE else batrArray.size - offset

                temp = cipher.doFinal(batrArray, offset, len)
                // 拼接解密结果
                bos.write(temp)
                // 偏移量后移
                offset += len
            }
    
            bos.close()
            // 转换为明文字符串返回
            return String(bos.toByteArray())
        }

## 消息摘要

* md5、sha1、sha256

* ```kotlin
  class MessageDigestUtil {
  
      fun md5(input: String): String {
          val digest = MessageDigest.getInstance("MD5")
          val result = digest.digest(input.toByteArray())
          return toHex(result)
      }
      fun sha1(input: String): String {
          val digest = MessageDigest.getInstance("SHA-1")
          val result = digest.digest(input.toByteArray())
          return toHex(result)
      }
      fun sha256(input: String): String {
          val digest = MessageDigest.getInstance("SHA-256")
          val result = digest.digest(input.toByteArray())
          return toHex(result)
      }
  
     private fun toHex(byteArray: ByteArray): String {
          return with(StringBuffer()) {
              byteArray.forEach {
                  val hex = it.toInt() and 0xff
                  val hexstr = Integer.toHexString(hex)
                  if (hexstr.length == 1) append('0').append(hexstr)else this.append(hexstr)
              }
               toString()
          }
      }
  }

## 数字签名

* 数字签名就是非对称加密与消息摘要的组合

* ```kotlin
  object DigitalSignatureUtil {
      fun sign(data: String, privateKey: PrivateKey): String {
          val signature = Signature.getInstance("SHA256WithRSA")
          signature.initSign(privateKey)
          signature.update(data.toByteArray())
          return Base64.getEncoder().encodeToString(signature.sign())
      }
  
      fun verify(data: String, publicKey: PublicKey,sign: String): Boolean {
          val signature = Signature.getInstance("SHA256WithRSA")
          signature.initVerify(publicKey)
          signature.update(data.toByteArray())
          return signature.verify(Base64.getDecoder().decode(sign))
      }
  }
